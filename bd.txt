CREATE TABLE coffees (
  id PRIMARY KEY AUTOINCREMENT,
  name text NOT NULL,
  roaster text,
  country text NOT NULL,
  region text,
  producer text,
  variety text[],
  process text NOT NULL,
  roast_level smallint,
  roast_date date,
  roaster_notes text[],
  created_at timestamptz DEFAULT now()
);

CREATE TABLE brew_logs (
  id PRIMARY KEY AUTOINCREMENT,
  user_id NOT NULL REFERENCES users(id),
  coffee_id REFERENCES coffees(id),
  brewed_at timestamptz DEFAULT now(),
  method text,
  dose_g numeric,
  water_g numeric,
  grind_setting text,
  water_temp_c smallint,
  brew_time_sec smallint,
  notes text,
  created_at timestamptz DEFAULT now()
);

CREATE TABLE flavor_tags (
  id PRIMARY KEY AUTOINCREMENT,
  name      text UNIQUE NOT NULL,
  category  text NOT NULL,
  parent_id int REFERENCES flavor_tags(id)
);

CREATE TABLE user (
  id PRIMARY KEY AUTOINCREMENT,
  name VARCHAR (100)
  coffee_id REFERENCES coffees(id),
  status text DEFAULT 'suggested',
  created_at timestamptz DEFAULT now()
);
